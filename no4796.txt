package mentoring;

import java.util.Scanner;

public class no4796 {
	public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);

		int num = 1;

		while (true) {

			int L = scanner.nextInt(); // P일 중, L일 동안만 사용 가능
			int P = scanner.nextInt(); // P일 중, L일 동안만 사용 가능
			int V = scanner.nextInt(); // V일짜리 휴가

			if (L == 0 && P == 0 && V == 0) // L, P, V 모두 0을 입력받을 경우 프로그램 종료
				break;

			int maximumDate = V / P * L + (V % P > L ? L : V % P); // 최대날짜

			System.out.println("Case " + num + ": " + maximumDate);

			num++; // 다음 케이스 숫자 증가시켜주기 위해서 증감연산자 사용
		}
	}
}
